; PgBouncer User Authentication File
; Format: "username" "password"
; 
; For SCRAM-SHA-256, use: "username" "SCRAM-SHA-256$<iterations>:<salt>$<StoredKey>:<ServerKey>"
; Or use md5: "username" "md5<md5hash>"
; Or plaintext (not recommended): "username" "plaintext_password"
;
; Generate SCRAM hash with:
; SELECT rolname, rolpassword FROM pg_authid WHERE rolname = 'tracker';
;
; IMPORTANT: this file is NOT mounted by docker-compose.prod.yml (pgbouncer
; there uses env-driven AUTH_TYPE=scram-sha-256). It is only a template. If
; you switch pgbouncer to a config-file mount, replace the placeholder below
; with a GENERATED SCRAM-SHA-256 hash — never commit a real or default
; plaintext password (finding L15).
;
; "your_db_user" "SCRAM-SHA-256$4096:<salt>$<StoredKey>:<ServerKey>"
